25 static BLAKE2_INLINE uint64_t fBlaMka(uint64_t x, uint64_t y) {
26  const uint64_t m = UINT64_C(0xFFFFFFFF);
27  const uint64_t xy = (x & m) * (y & m);
28  return x + y + 2 * xy;
29 }
