90  inline void SIPROUND()
91  {
92  m_v[0] += m_v[1];
93  m_v[1] = rotlFixed(m_v[1], 13U);
94  m_v[1] ^= m_v[0];

158 void SipHash_Base<C,D,T_128bit>::Update(const byte *input, size_t length)
159 {
160  CRYPTOPP_ASSERT((input && length) || !length);
161  if (!length) return;
162 
163  if (m_idx)
164  {
165  size_t head = STDMIN(size_t(8U-m_idx), length);
166  memcpy(m_acc+m_idx, input, head);
167  m_idx += head; input += head; length -= head;
168 
169  if (m_idx == 8)
170  {
171  word64 m = GetWord<word64>(true, LITTLE_ENDIAN_ORDER, m_acc);
172  m_v[3] ^= m;
173  for (unsigned int i = 0; i < C; ++i)
174  SIPROUND();
175 
176  m_v[0] ^= m;
177  m_b[0] += 8;
178 
179  m_idx = 0;
180  }
181  }
182 
183  while (length >= 8)
184  {
185  word64 m = GetWord<word64>(false, LITTLE_ENDIAN_ORDER, input);
186  m_v[3] ^= m;
187  for (unsigned int i = 0; i < C; ++i)
188  SIPROUND();
189 
190  m_v[0] ^= m;
191  m_b[0] += 8;
192 
193  input += 8;
194  length -= 8;
195  }
196 
197  CRYPTOPP_ASSERT(length < 8);
198  size_t tail = length % 8;
199  if (tail)
200  {
201  memcpy(m_acc+m_idx, input, tail);
202  m_idx += tail;
203  }
204 }

206 template <unsigned int C, unsigned int D, bool T_128bit>
207 void SipHash_Base<C,D,T_128bit>::TruncatedFinal(byte *digest, size_t digestSize)
208 {
209  CRYPTOPP_ASSERT(digest); // Pointer is valid
210 
211  ThrowIfInvalidTruncatedSize(digestSize);
212 
213  // The high octet holds length and is digested mod 256
214  m_b[0] += m_idx; m_b[0] <<= 56U;
215  switch (m_idx)
216  {
217  case 7:
218  m_b[0] |= ((word64)m_acc[6]) << 48;
219  // fall through
220  case 6:
221  m_b[0] |= ((word64)m_acc[5]) << 40;
222  // fall through
223  case 5:
224  m_b[0] |= ((word64)m_acc[4]) << 32;
225  // fall through
226  case 4:
227  m_b[0] |= ((word64)m_acc[3]) << 24;
228  // fall through
229  case 3:
230  m_b[0] |= ((word64)m_acc[2]) << 16;
231  // fall through
232  case 2:
233  m_b[0] |= ((word64)m_acc[1]) << 8;
234  // fall through
235  case 1:
236  m_b[0] |= ((word64)m_acc[0]);
237  // fall through
238  case 0:
239  break;
240  }
241 
242  m_v[3] ^= m_b[0];
243 
244  for (unsigned int i=0; i<C; i++)
245  SIPROUND();
246 
247  m_v[0] ^= m_b[0];
248 
249  if (T_128bit)
250  m_v[2] ^= 0xee;
251  else
252  m_v[2] ^= 0xff;
253 
254  for (unsigned int i=0; i<D; i++)
255  SIPROUND();
256 
257  m_b[0] = m_v[0] ^ m_v[1] ^ m_v[2] ^ m_v[3];
258  m_b[0] = ConditionalByteReverse(LITTLE_ENDIAN_ORDER, m_b[0]);
259 
260  if (T_128bit)
261  {
262  m_v[1] ^= 0xdd;
263  for (unsigned int i = 0; i<D; ++i)
264  SIPROUND();
265 
266  m_b[1] = m_v[0] ^ m_v[1] ^ m_v[2] ^ m_v[3];
267  m_b[1] = ConditionalByteReverse(LITTLE_ENDIAN_ORDER, m_b[1]);
268  }
269 
270  memcpy_s(digest, digestSize, m_b.begin(), STDMIN(digestSize, (size_t)SipHash_Info<T_128bit>::DIGESTSIZE));
271  Restart();
272 }
