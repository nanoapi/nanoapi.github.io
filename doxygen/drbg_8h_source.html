378 template <typename HASH, unsigned int STRENGTH, unsigned int SEEDLENGTH>
379 void Hash_DRBG<HASH, STRENGTH, SEEDLENGTH>::DRBG_Instantiate(const byte* entropy, size_t entropyLength, const byte* nonce, size_t nonceLength,
380  const byte* personalization, size_t personalizationLength)
381 {
382  // SP 800-90A, 8.6.3: The entropy input shall have entropy that is equal to or greater than the security
383  // strength of the instantiation. Additional entropy may be provided in the nonce or the optional
384  // personalization string during instantiation, or in the additional input during reseeding and generation,
385  // but this is not required and does not increase the "official" security strength of the DRBG
386  // instantiation that is recorded in the internal state.
387  CRYPTOPP_ASSERT(entropyLength >= MINIMUM_ENTROPY);
388  if (entropyLength < MINIMUM_ENTROPY)
389  throw NIST_DRBG::Err("Hash_DRBG", "Insufficient entropy during instantiate");
390 
391  // SP 800-90A, Section 9, says we should throw if we have too much entropy, too large a nonce,
392  // or too large a persoanlization string. We warn in Debug builds, but do nothing in Release builds.
393  CRYPTOPP_ASSERT(entropyLength <= MAXIMUM_ENTROPY);
394  CRYPTOPP_ASSERT(nonceLength <= MAXIMUM_NONCE);
395  CRYPTOPP_ASSERT(personalizationLength <= MAXIMUM_PERSONALIZATION);
396 
397  const byte zero = 0;
398  SecByteBlock t1(SEEDLENGTH), t2(SEEDLENGTH);
399  Hash_Update(entropy, entropyLength, nonce, nonceLength, personalization, personalizationLength, NULLPTR, 0, t1, t1.size());
400  Hash_Update(&zero, 1, t1, t1.size(), NULLPTR, 0, NULLPTR, 0, t2, t2.size());
401 
402  m_v.swap(t1); m_c.swap(t2);
403  m_reseed = 1;
404 }

434 template <typename HASH, unsigned int STRENGTH, unsigned int SEEDLENGTH>
435 void Hash_DRBG<HASH, STRENGTH, SEEDLENGTH>::Hash_Generate(const byte* additional, size_t additionaLength, byte *output, size_t size)
436 {
437  // Step 1
438  if (static_cast<word64>(m_reseed) >= static_cast<word64>(MaxRequestBeforeReseed()))
439  throw NIST_DRBG::Err("Hash_DRBG", "Reseed required");
440 
441  if (size > MaxBytesPerRequest())
442  throw NIST_DRBG::Err("Hash_DRBG", "Request size exceeds limit");
443 
444  // SP 800-90A, Section 9, says we should throw if we have too much entropy, too large a nonce,
445  // or too large a persoanlization string. We warn in Debug builds, but do nothing in Release builds.
446  CRYPTOPP_ASSERT(additionaLength <= MAXIMUM_ADDITIONAL);
447 
448  // Step 2
449  if (additional && additionaLength)
450  {
451  HASH hash;
452  const byte two = 2;
453  SecByteBlock w(HASH::DIGESTSIZE);
454 
455  hash.Update(&two, 1);
456  hash.Update(m_v, m_v.size());
457  hash.Update(additional, additionaLength);
458  hash.Final(w);
459 
460  CRYPTOPP_ASSERT(SEEDLENGTH >= HASH::DIGESTSIZE);
461  int carry=0, j=HASH::DIGESTSIZE-1, i=SEEDLENGTH-1;
462  while (j>=0)
463  {
464  carry = m_v[i] + w[j] + carry;
465  m_v[i] = static_cast<byte>(carry);
466  i--; j--; carry >>= 8;
467  }
468  while (i>=0)
469  {
470  carry = m_v[i] + carry;
471  m_v[i] = static_cast<byte>(carry);
472  i--; carry >>= 8;
473  }
474  }
475 
476  // Step 3
477  {
478  HASH hash;
479  SecByteBlock data(m_v);
480 
481  while (size)
482  {
483  hash.Update(data, data.size());
484  size_t count = STDMIN(size, (size_t)HASH::DIGESTSIZE);
485  hash.TruncatedFinal(output, count);
486 
487  IncrementCounterByOne(data, static_cast<unsigned int>(data.size()));
488  size -= count; output += count;
489  }
490  }
491 
492  // Steps 4-7
493  {
494  HASH hash;
495  const byte three = 3;
496  SecByteBlock h(HASH::DIGESTSIZE);
497 
498  hash.Update(&three, 1);
499  hash.Update(m_v, m_v.size());
500  hash.Final(h);
501 
502  CRYPTOPP_ASSERT(SEEDLENGTH >= HASH::DIGESTSIZE);
503  CRYPTOPP_ASSERT(HASH::DIGESTSIZE >= sizeof(m_reseed));
504  int carry=0, k=sizeof(m_reseed)-1, j=HASH::DIGESTSIZE-1, i=SEEDLENGTH-1;
505 
506  // Using Integer class slows things down by about 8 cpb.
507  // Using word128 and word64 benefits the first loop only by about 2 cpb.
508 #if defined(CRYPTOPP_WORD128_AVAILABLE)
509  byte* p1 = m_v.begin()+SEEDLENGTH-8;
510  byte* p2 = m_c.begin()+SEEDLENGTH-8;
511  byte* p3 = h.begin()+HASH::DIGESTSIZE-8;
512 
513  const word64 w1 = GetWord<word64>(false, BIG_ENDIAN_ORDER, p1);
514  const word64 w2 = GetWord<word64>(false, BIG_ENDIAN_ORDER, p2);
515  const word64 w3 = GetWord<word64>(false, BIG_ENDIAN_ORDER, p3);
516  const word64 w4 = m_reseed;
517 
518  word128 r = static_cast<word128>(w1) + w2 + w3 + w4;
519  PutWord(false, BIG_ENDIAN_ORDER, p1, static_cast<word64>(r));
520  i -= 8; j -= 8; k=0; carry = static_cast<int>(r >> 64);
521 
522  // The default implementation and a couple of others cause a crash in
523  // VS2005, VS2008 and VS2105. This seems to work with all MS compilers.
524 #elif defined(CRYPTOPP_MSC_VERSION)
525  byte* p1 = m_v.begin()+SEEDLENGTH-8;
526  byte* p2 = m_c.begin()+SEEDLENGTH-8;
527  byte* p3 = h.begin()+HASH::DIGESTSIZE-8;
528 
529  const word64 w1 = GetWord<word64>(false, BIG_ENDIAN_ORDER, p1);
530  const word64 w2 = GetWord<word64>(false, BIG_ENDIAN_ORDER, p2);
531  const word64 w3 = GetWord<word64>(false, BIG_ENDIAN_ORDER, p3);
532  const word64 w4 = m_reseed;
533 
534  const word64 r1 = (w1 & 0xffffffff) + (w2 & 0xffffffff) + (w3 & 0xffffffff) + (w4 & 0xffffffff);
535  carry = static_cast<int>(r1 >> 32);
536  const word64 r2 = (w1 >> 32) + (w2 >> 32) + (w3 >> 32) + (w4 >> 32) + carry;
537  carry = static_cast<int>(r2 >> 32);
538 
539  const word64 r = (r2 << 32) + (r1 & 0xffffffff);
540  PutWord(false, BIG_ENDIAN_ORDER, p1, r);
541  i -= 8; j -= 8; k=0;
542 
543  // Default implementation, but slower on some machines.
544 #else
545  while (k>=0)
546  {
547  carry = m_v[i] + m_c[i] + h[j] + GetByte<word64>(BIG_ENDIAN_ORDER, m_reseed, k) + carry;
548  m_v[i] = static_cast<byte>(carry);
549  i--; j--; k--; carry >>= 8;
550  }
551 #endif
552 
553  while (j>=0)
554  {
555  carry = m_v[i] + m_c[i] + h[j] + carry;
556  m_v[i] = static_cast<byte>(carry);
557  i--; j--; carry >>= 8;
558  }
559 
560  while (i>=0)
561  {
562  carry = m_v[i] + m_c[i] + carry;
563  m_v[i] = static_cast<byte>(carry);
564  i--; carry >>= 8;
565  }
566 
567  // CRYPTOPP_WORD128_AVAILABLE causes -Wunused-but-set-variable
568  CRYPTOPP_UNUSED(k);
569  }
570 
571  m_reseed++;
572 }

575 template <typename HASH, unsigned int STRENGTH, unsigned int SEEDLENGTH>
576 void Hash_DRBG<HASH, STRENGTH, SEEDLENGTH>::Hash_Update(const byte* input1, size_t inlen1, const byte* input2, size_t inlen2,
577  const byte* input3, size_t inlen3, const byte* input4, size_t inlen4, byte* output, size_t outlen)
578 {
579  HASH hash;
580  byte counter = 1;
581  word32 bits = ConditionalByteReverse(BIG_ENDIAN_ORDER, static_cast<word32>(outlen*8));
582 
583  while (outlen)
584  {
585  hash.Update(&counter, 1);
586  hash.Update(reinterpret_cast<const byte*>(&bits), 4);
587 
588  if (input1 && inlen1)
589  hash.Update(input1, inlen1);
590  if (input2 && inlen2)
591  hash.Update(input2, inlen2);
592  if (input3 && inlen3)
593  hash.Update(input3, inlen3);
594  if (input4 && inlen4)
595  hash.Update(input4, inlen4);
596 
597  size_t count = STDMIN(outlen, (size_t)HASH::DIGESTSIZE);
598  hash.TruncatedFinal(output, count);
599 
600  output += count; outlen -= count;
601  counter++;
602  }
603 }

650  CRYPTOPP_ASSERT(additionaLength <= MAXIMUM_ADDITIONAL);
651 
652  HMAC_Update(entropy, entropyLength, additional, additionaLength, NULLPTR, 0);
653  m_reseed = 1;
654 }

657 template <typename HASH, unsigned int STRENGTH, unsigned int SEEDLENGTH>
658 void HMAC_DRBG<HASH, STRENGTH, SEEDLENGTH>::HMAC_Generate(const byte* additional, size_t additionaLength, byte *output, size_t size)
659 {
660  // Step 1
661  if (static_cast<word64>(m_reseed) >= static_cast<word64>(MaxRequestBeforeReseed()))
662  throw NIST_DRBG::Err("HMAC_DRBG", "Reseed required");
663 
664  if (size > MaxBytesPerRequest())
665  throw NIST_DRBG::Err("HMAC_DRBG", "Request size exceeds limit");
666 
667  // SP 800-90A, Section 9, says we should throw if we have too much entropy, too large a nonce,
668  // or too large a persoanlization string. We warn in Debug builds, but do nothing in Release builds.
669  CRYPTOPP_ASSERT(additionaLength <= MAXIMUM_ADDITIONAL);
670 
671  // Step 2
672  if (additional && additionaLength)
673  HMAC_Update(additional, additionaLength, NULLPTR, 0, NULLPTR, 0);
674 
675  // Step 3
676  HMAC<HASH> hmac;
677  hmac.SetKey(m_k, m_k.size());
678 
679  while (size)
680  {
681  hmac.Update(m_v, m_v.size());
682  hmac.TruncatedFinal(m_v, m_v.size());
683 
684  size_t count = STDMIN(size, (size_t)HASH::DIGESTSIZE);
685  memcpy(output, m_v, count);
686 
687  size -= count; output += count;
688  }
689 
690  HMAC_Update(additional, additionaLength, NULLPTR, 0, NULLPTR, 0);
691  m_reseed++;
692 }

695 template <typename HASH, unsigned int STRENGTH, unsigned int SEEDLENGTH>
696 void HMAC_DRBG<HASH, STRENGTH, SEEDLENGTH>::HMAC_Update(const byte* input1, size_t inlen1, const byte* input2, size_t inlen2, const byte* input3, size_t inlen3)

701  // Step 1
702  hmac.SetKey(m_k, m_k.size());

705

710  if (input3 && inlen3)

714

717  hmac.Update(m_v, m_v.size());
718 
719  hmac.TruncatedFinal(m_v, m_v.size());

721  // Step 3

724

727  hmac.Update(m_v, m_v.size());
728  hmac.Update(&one, 1);
729 
730  if (input1 && inlen1)
731  hmac.Update(input1, inlen1);
732  if (input2 && inlen2)
733  hmac.Update(input2, inlen2);
734  if (input3 && inlen3)
735  hmac.Update(input3, inlen3);

741  hmac.Update(m_v, m_v.size());
