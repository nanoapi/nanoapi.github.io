241 void AutoSeededX917RNG<BLOCK_CIPHER>::Reseed(bool blocking, const byte *input, size_t length)
242 {
243  SecByteBlock seed(BLOCK_CIPHER::BLOCKSIZE + BLOCK_CIPHER::DEFAULT_KEYLENGTH);
244  const byte *key;
245  do
246  {
247  OS_GenerateRandomBlock(blocking, seed, seed.size());
248  if (length > 0)
249  {
250  SHA256 hash;
251  hash.Update(seed, seed.size());
252  hash.Update(input, length);
253  hash.TruncatedFinal(seed, UnsignedMin(hash.DigestSize(), seed.size()));
254  }
255  key = seed + BLOCK_CIPHER::BLOCKSIZE;
256  } // check that seed and key don't have same value
257  while (memcmp(key, seed, STDMIN((unsigned int)BLOCK_CIPHER::BLOCKSIZE, (unsigned int)BLOCK_CIPHER::DEFAULT_KEYLENGTH)) == 0);
258 
259  Reseed(key, BLOCK_CIPHER::DEFAULT_KEYLENGTH, seed, NULLPTR);
260 }
