433  struct RegisterOutput

435  RegisterOutput(byte *output, const byte *input, CipherDir dir)
436  : m_output(output), m_input(input), m_dir(dir) {}
437 
441  inline RegisterOutput& operator()(WordType &registerWord)
442  {
443  CRYPTOPP_ASSERT(IsAligned<WordType>(m_output));
444  CRYPTOPP_ASSERT(IsAligned<WordType>(m_input));
445 
446  if (!NativeByteOrderIs(B::ToEnum()))
447  registerWord = ByteReverse(registerWord);
448 
449  if (m_dir == ENCRYPTION)
450  {
451  if (m_input == NULLPTR)
452  {
453  CRYPTOPP_ASSERT(m_output == NULLPTR);
454  }
455  else
456  {
457  WordType ct = *(const WordType *)m_input ^ registerWord;
458  registerWord = ct;
459  *(WordType*)m_output = ct;
460  m_input += sizeof(WordType);
461  m_output += sizeof(WordType);
462  }
463  }
464  else
465  {
466  WordType ct = *(const WordType *)m_input;
467  *(WordType*)m_output = registerWord ^ ct;
468  registerWord = ct;
469  m_input += sizeof(WordType);
470  m_output += sizeof(WordType);
471  }
472 
473  // registerWord is left unreversed so it can be xor-ed with further input
474 
475  return *this;
476  }
